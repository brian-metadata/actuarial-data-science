    # Label segments only if wide enough for readability
    if width > 0.08:
        pct_label = f"{(width*100):.0f}%\n{label.split(' ')[0]}"
        plt.text(left + width/2, 0, pct_label, ha='center', va='center', fontsize=10, color='white')
    left += width
